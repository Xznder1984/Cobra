Language Reference

Complete reference for the Cobra programming language syntax and semantics.

Comments

# This is a comment (hash style)
// This is also a comment (C++ style)
/* Block comments
   span multiple lines */

Variables

Immutable (let)

let x = 5
let name = "Cobra"

Mutable (let mut)

let mut count = 0
count = count + 1

Explicit type annotations

let x: int = 5
let name: string = "Cobra"

Constants

const MAX_SIZE = 1024

Functions

Function declarations

fn greet(name: string) {
    println("Hello, " + name)
}

fn add(a: int, b: int) -> int {
    return a + b
}

Expression body

fn add(a: int, b: int) -> int => a + b

Main function (entry point)

fn main() {
    println("Hello, World!")
}

Default parameters

fn greet(name: string = "World") {
    println("Hello, " + name)
}

Control Flow

if / elif / else

if x > 0 {
    println("positive")
} elif x < 0 {
    println("negative")
} else {
    println("zero")
}

If expressions

let status = if x > 0 { "positive" } else { "non-positive" }

While loops

let mut i = 0
while i < 10 {
    println(i)
    i = i + 1
}

For loops

for i in 0..10 {
    println(i)
}

for item in list {
    println(item)
}

Match

match value {
    1 => println("one"),
    2 => println("two"),
    _ => println("other"),
}

Structs

struct Point {
    x: int,
    y: int,
}

let p = Point { x: 10, y: 20 }
println(p.x)

Classes

class Animal {
    name: string

    fn speak(self) {
        println(self.name + " makes a sound")
    }
}

let dog = Animal { name: "Dog" }
dog.speak()

Enums

enum Color {
    Red,
    Green,
    Blue,
}

Traits

trait Drawable {
    fn draw(self)
}

impl Circle : Drawable {
    fn draw(self) {
        println("Drawing a circle")
    }
}

Pattern Matching

match value {
    0 => println("none"),
    1 | 2 => println("few"),
    3..=10 => println("some"),
    _ => println("many"),
}

match x {
    _ if x > 0 => println("positive"),
    _ => println("non-positive"),
}

Error Handling

fn divide(a: int, b: int) -> Result<int, string> {
    if b == 0 {
        return Err("division by zero")
    }
    return Ok(a / b)
}

let result = try divide(10, 2)

Concurrency

async fn fetch_data(url: string) -> string {
    return await http.get(url)
}

fn main() {
    let future = async fetch_data("https://example.com")
    let data = await future
}

Other Features

Defer

fn read_file(path: string) {
    let file = open(path)
    defer close(file)
    // file is automatically closed when function returns
}

Unsafe blocks

unsafe {
    // raw pointer operations, FFI calls, etc.
}

Type aliases

type Age = int
type Callback = fn(int) -> bool
